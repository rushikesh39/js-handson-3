Find the number of digits 

        const Find_Digits = (N) => 
        {
             let count=0;
             while(N!==0){
               count++;
               N=Math.floor(N/10);
               
             }
             return count;
        };

        const findSum = (n) => {

Check whether a Number is a prime or not. 

        const Prime_Check = (N) => 
        {
             let factor=0;
             for(i=1;i<=N;i++){
               if(N%i===0){
                 factor++
               }
             }
             if(factor===2){
               return "YES";
             }
             else{
               return "NO";
             }
            
        };
